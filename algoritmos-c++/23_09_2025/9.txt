Algoritmo Factura
	Definir imp, iva, simp, siva, d, tot Como Real
	simp <- 0
	siva <- 0
	Repetir
		Escribir "Ingrese el importe (0 para terminar):"
		Leer imp
		Si imp <> 0 Entonces
			Escribir "Ingrese el IVA (4, 7 o 16):"
			Leer iva
			Si iva=4 O iva=7 O iva=16 Entonces
				simp <- simp + imp
				siva <- siva + imp*iva/100
			SiNo
				Escribir "intente de nuevo"
			FinSi
		FinSi
	Hasta Que imp=0
	Si simp < 1000 Entonces
		d <- 0
	SiNo
		Si simp < 10000 Entonces
			d <- 5
		SiNo
			d <- 10
		FinSi
	FinSi
	simp <- simp - simp*d/100
	siva <- siva - siva*d/100
	tot <- simp + siva
	Escribir "Importe:", simp
	Escribir "IVA:", siva
	Escribir "TOTAL:", tot
FinAlgoritmo
