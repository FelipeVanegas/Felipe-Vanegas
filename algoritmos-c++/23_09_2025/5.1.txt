Algoritmo Suma_Impares
	Definir a, b, c, d Como Entero
	Escribir 'Ingrese la cantidad de números impares que desea sumar:'
	Leer a
	c <- 0
	d <- 0
	Mientras d<a Hacer
		Escribir 'Ingrese un número impar:'
		Leer b
		Si b MOD 2<>0 Entonces
			c <- c+b
			d <- d+1
		SiNo
			Escribir 'No es impar intente de nuevo.'
		FinSi
	FinMientras
	Escribir 'La suma de los ', a, ' impares es: ', c
FinAlgoritmo
